\section{正定质量粒子因果场\(\dagger\)}
一般形式场写为：\begin{equation*}
    \Psi_\ell(x) =(\frac{1}{2\pi})^\frac{3}{2}\sum_{\sigma}\int d^3p \, \left\{\xi_\ell u_\ell(\boldsymbol{p},\sigma)a^{c\dagger}(\boldsymbol{p},\sigma) e^{-ipx} + \lambda_\ell v_\ell(\boldsymbol{p},\sigma) a(\boldsymbol{p},\sigma)e^{ipx} \right\}
\end{equation*}
若理论本身无荷，就有$a^{\dagger}(\boldsymbol{p},\sigma)=a^{c\dagger}(\boldsymbol{p},\sigma)=a^{\dagger}(\boldsymbol{p},\sigma)$；系数$u_\ell(\boldsymbol{p},\sigma)$和$v_\ell(\boldsymbol{p},\sigma)$的变换是
\begin{equation*}
    \begin{aligned}
            &\sum_{\bar{\sigma}}
            u_{\ell}(\boldsymbol{p},\bar{\sigma})
            \sqrt{\frac{p^0}{(\Lambda p)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n}
            \!\left\{W^{-1}(\Lambda,p)\right\} =
            \sum_{\bar{\ell}}
            \mathscr{D}_{\ell\bar{\ell}}(\Lambda^{-1})
            u_{\bar{\ell}}(\Lambda \boldsymbol{p},\sigma)\\
            &\sum_{\bar{\sigma}}
            v_{\ell}(\boldsymbol{p},\bar{\sigma})
            \sqrt{\frac{p^0}{(\Lambda p)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n*}
            \!\left\{W^{-1}(\Lambda,p)\right\} =
            \sum_{\bar{\ell}}
            \mathscr{D}_{\ell\bar{\ell}}(\Lambda^{-1})
            v_{\bar{\ell}}(\Lambda \boldsymbol{p},\sigma)
            \end{aligned}
\end{equation*}
\begin{center}
    \vspace{2ex}
    \large\textbf{***}
    \vspace{2ex}
\end{center}
通过 $SU(2)_L \times SU(2)_R$ 代数构建Lorentz群的有限维表示后。指标$\ell$在这里被替换为一对指标$a$、$b$：所以一般形式场可以写为：\begin{equation}
    \Psi_{ab}(x) =(\frac{1}{2\pi})^\frac{3}{2}\sum_{\sigma}\int d^3p \, \left\{\xi_{ab} u_{ab}(\boldsymbol{p},\sigma)a^{c\dagger}(\boldsymbol{p},\sigma) e^{-ipx} + \lambda_{ab} v_{ab}(\boldsymbol{p},\sigma) a(\boldsymbol{p},\sigma)e^{ipx} \right\}
\end{equation}
以及系数函数的选择变为：
\begin{subequations}
    \begin{align}
            &\sum_{\bar{\sigma}}
            u_{ab}(\boldsymbol{p},\bar{\sigma})
            \sqrt{\frac{p^0}{(\Lambda p)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n}
            \!\left\{W^{-1}(\Lambda,p)\right\} =
            \sum_{\bar{a}\bar{b}}
            \mathscr{D}_{{ab};\bar{a}\bar{{b}}}(\Lambda^{-1})
            u_{\bar{a}\bar{b}}(\Lambda \boldsymbol{p},\sigma)
            \\
             &\sum_{\bar{\sigma}}
            v_{ab}(\boldsymbol{p},\bar{\sigma})
            \sqrt{\frac{p^0}{(\Lambda p)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n*}
            \!\left\{W^{-1}(\Lambda,p)\right\} =
            \sum_{\bar{a}\bar{b}}
            \mathscr{D}_{{ab};\bar{a}\bar{{b}}}(\Lambda^{-1})
            v_{\bar{a}\bar{b}}(\Lambda \boldsymbol{p},\sigma)
            \end{align}
\end{subequations}
首先确定系数的选择；考虑剩下的两组Poincar\'e变换，依次考察$\operatorname{Boost}$和$\operatorname{Rotation}$；
\vspace{2ex}
\begin{center}
    \large\textbf{boost}  
\end{center}
\vspace{2ex}
令$p^\mu=(p^0,\boldsymbol{p})$，也就是任意动量系；选择$\Lambda=L^{-1}(p)$是任意系到静止系标准Lorentz变换，$\Lambda p=k=(m,0,0,0)$，所以：
\[\begin{aligned}
    &W(\Lambda,p)=1\to\sum_{\bar{\sigma}}D_{\sigma \bar{\sigma}}^{j_n}\!\left\{W^{-1}(\Lambda,p)\right\} =1\\
    &\qquad(\Lambda p)^0\to m^0
\end{aligned}\]
这种情况下，系数函数表述为：
\begin{subequations}
    \begin{align}\label{eq:8.38a}
       &\sum_{\bar{\sigma}}u_{ab}(\boldsymbol{p},\bar{\sigma})\delta_{\sigma\bar{\sigma}}\sqrt{\frac{p^0}{m}}
        =\sum_{\bar{a}\bar{b}}\mathscr{D}_{ab;\bar{a}\bar{b}}(L(p))u_{\bar{a}\bar{b}}(0,\sigma)\\
       &\sum_{\bar{\sigma}}v_{ab}(\boldsymbol{p},\bar{\sigma})\delta_{\sigma\bar{\sigma}}\sqrt{\frac{p^0}{m}}
        =\sum_{\bar{a}\bar{b}}\mathscr{D}_{ab;\bar{a}\bar{b}}(L())v_{\bar{a}\bar{b}}(0,\sigma)
\end{align}
\end{subequations}
\vspace{2ex}
\begin{center}
   \large \textbf{rotation}  
\end{center}
\vspace{2ex}
令$p^\mu=(p^0,0,0,0)$，也就是静止系；选择$\Lambda=R(\theta)$也就是在静止系下保持$p^\mu=(p^0,0,0,0)$不变的旋转，那么有$R(p)=p$以及$\Lambda p=p$，所以：
\[W(\Lambda,p)=R\]
\color{blue}由于$\boldsymbol{J}=\boldsymbol{A}+\boldsymbol{B}$，$Winger$的转动表征为总的自旋$\boldsymbol{J}$,
$$ D_{\sigma\bar{\sigma}}^{(j)}(R^{-1}) = \left[ \exp\left( +i \boldsymbol{\theta}\cdot\boldsymbol{J}^{(j)} \right) \right]_{\sigma\bar{\sigma}} $$
场的自旋拆为$\boldsymbol{A}+\boldsymbol{B}$
\[\mathscr{D}_{{ab};\bar{a}\bar{{b}}}(R^{-1})=\left[\exp\left( +i \boldsymbol{\theta}\cdot\boldsymbol{A}+i \boldsymbol{\theta}\cdot\boldsymbol{B}\right) \right]_{{ab};\bar{a}\bar{{b}}}\]\normalcolor
在这种分解下系数函数的选择\footnote{这个方程也叫做交织方程}变为：
\begin{subequations}\label{u_0}
    \begin{align}
            &\sum_{\bar{\sigma}} 
            \left[ \exp\left( -i \boldsymbol{\theta}\cdot\boldsymbol{J}^{(j)} \right) \right]_{\sigma\bar{\sigma}}u_{ab}(0,\bar{\sigma}) =
            \sum_{\bar{a}\bar{b}}
            \left[\exp\left(  +i \boldsymbol{\theta}\cdot\boldsymbol{A}+i \boldsymbol{\theta}\cdot\boldsymbol{B} \right) \right]_{{ab};\bar{a}\bar{{b}}}
            u_{\bar{a}\bar{b}}(0,\sigma)
            \\
            &\sum_{\bar{\sigma}} 
            \left[ \exp\left( +i \boldsymbol{\theta}\cdot\boldsymbol{J}^{(j)} \right) \right]_{\sigma\bar{\sigma}}v_{ab}(0,\bar{\sigma}) =
            \sum_{\bar{a}\bar{b}}
            \left[\exp\left(  +i \boldsymbol{\theta}\cdot\boldsymbol{A}+i \boldsymbol{\theta}\cdot\boldsymbol{B} \right) \right]_{{ab};\bar{a}\bar{{b}}}
            v_{\bar{a}\bar{b}}(0,\sigma)
            \end{align}
\end{subequations}
\color{blue}下面只需要在静止系中确定系数的形式；之后通过选择一个标准的$boost$来获得动量。\normalcolor
\begin{center}
    \vspace{2ex}
    \large\text{系数$u(0,\sigma)$和$v(0,\sigma)$和角动量加法}
\end{center}
\vspace{2ex}
交织方程(\refeq{u_1})在$\boldsymbol{\theta}$ 在 $\boldsymbol{\theta}=0$ 处展开至一阶对比两边 $\boldsymbol{\theta}$ 的系数，，到静止系下系数函数的代数关系变为
\begin{subequations}
    \begin{align}
       -\sum_{\bar{\sigma}} u_{ab}(0,\bar{\sigma}) \left[ \boldsymbol{J}^{(j)} \right]_{\bar{\sigma}\sigma} &= 
       \sum_{\bar{a}\bar{b}} \left[ \mathbf{A}_{a\bar{a}} \delta_{b\bar{b}} + \delta_{a\bar{a}} \mathbf{B}_{b\bar{b}} \right] u_{\bar{a}\bar{b}}(0,\sigma)
        \\
        +\sum_{\bar{\sigma}} v_{ab}(0,\bar{\sigma}) \left[ \boldsymbol{J}^{(j)} \right]_{\bar{\sigma}\sigma} &= 
        \sum_{\bar{a}\bar{b}} \left[ \mathbf{A}_{a\bar{a}} \delta_{b\bar{b}} + \delta_{a\bar{a}} \mathbf{B}_{b\bar{b}} \right] v_{\bar{a}\bar{b}}(0,\sigma)
    \end{align}
\end{subequations}
\color{blue}这是一个极其深刻的物理方程。这是两个角动量算符的直积之和：$\boldsymbol{J}^{(A)} \otimes \mathbb{I} + \mathbb{I} \otimes \boldsymbol{J}^{(B)}$；而等式左边则是单一的总角动量算符 $\boldsymbol{J}^{(j)}$。实际上,系数 $v_{ab}(0, \sigma)$ 的等式，本质上就是角动量相加方法\normalcolor。为了清晰地展现这一点，首先明确系统的两种物理表象：
\begin{enumerate}
    \item 非耦合表象：
    场的洛伦兹张量指标 $(a,b)$ 属于表示空间 $V_A \otimes V_B$。如果我们将 $A$ 和 $B$ 视为两个独立的子系统，，对应的正交归一基底为：
    \begin{equation}
        |A,a; B,b\rangle \equiv |A,a\rangle \otimes |B,b\rangle, \quad a = -A, \dots, A, \quad b = -B, \dots, B.
    \end{equation}
    这组基底构成了系统的完备集，满足 $\sum_{a,b} |A,a; B,b\rangle \langle A,a; B,b| = \mathbb{I}$。

    \item \textbf{耦合表象}：
    对于静止参考系下的物理单粒子态，必须在旋转下按确定自旋 $j$ 的不可约表示变换。我们用总角动量基底来标记物理态：
    \begin{equation}
        |j, \bar{\sigma}\rangle, \quad \bar{\sigma} = -j, \dots, j.
    \end{equation}
\end{enumerate}
在耦合表象中插入非耦合表象的完备基底，就可以表达出$|j, \bar{\sigma}\rangle$；
\begin{equation}
    |j, \bar{\sigma}\rangle = \mathbb{I} \, |j, \bar{\sigma}\rangle = \sum_{a,b} |A,a; B,b\rangle \langle A,a; B,b | j, \bar{\sigma}\rangle.
    \label{eq:QM_expansion}
\end{equation}
这里的展开系数 $\langle A,a; B,b | j, \bar{\sigma}\rangle$，，定义就是连接这两个空间的 \textbf{Clebsch-Gordan 系数}
\vspace{2ex}\\
系数 $u_{ab}(0, \bar{\sigma})$ 的物理意义;是将物理自旋态“投影”到具体的场张量分量上。不严谨但形象的说，$u_{ab}(0, \bar{\sigma})$ 充当了态矢量的展开系数：
\begin{equation}
    |j, \bar{\sigma}\rangle \sim \sum_{a,b} |A,a; B,b\rangle \, v_{ab}(0, \bar{\sigma}).
    \label{eq:Field_expansion}
\end{equation}
比较式 \eqref{eq:QM_expansion} 与式 \eqref{eq:Field_expansion}，结合我们刚刚推导出的交织方程：方程要求 $u_{ab}(0, \bar{\sigma})$ 必须满足总角动量守恒的变换规则，与 C-G 系数所满足的代数方程是完全一致的。
\vspace{2ex}\\
从群表示论的严格角度来看， $SU(2)$ 的角动量加法 
\begin{equation}
V^{(A)} \otimes V^{(B)}
=
\bigoplus_{j=|A-B|}^{A+B}
V^{(j)} .
\label{eq:CG_decomposition}
\end{equation} 中，一个自旋 $j$ 的不可约表示在直和分解中只出现一次。这意味着：够满足这种旋转协变性的基底变换矩阵是\textbf{唯一确定的}（Schur 引理的推论）。
这种唯一性说明了 $u_{ab}(0, \bar{\sigma})$是正比于C-G 系数：
\begin{equation}
    v_{ab}(0, \bar{\sigma}) = N \langle A,a; B,b \,|\, j, \bar{\sigma}\rangle.
    \label{eq:uCG_proportional}
\end{equation}
这里 $N$ 是一个与群论旋转无关的整体常数。约定取 $N = \frac{1}{\sqrt{2m}}$，，终得到
\begin{equation}
    v_{ab}(0, \bar{\sigma}) = \frac{1}{\sqrt{2m}} \langle A,a; B,b \,|\, j, \bar{\sigma}\rangle.
\end{equation}
利用角动量态在反演下的相因子变换性质，及 C-G 系数的内在对称性，可以推导出 $v_{ab}$ 对应于物理自旋方向相反的态，带有一个标准的 Condon-Shortley 相位：
\begin{equation}
    u_{ab}(0,\bar{\sigma}) = \xi^{AB} \, (-1)^{j-\bar{\sigma}} \langle A, a ; B, b \,|\, j, -\bar{\sigma} \rangle
    \label{eq:v_CG}
\end{equation}
其中 $\xi^{AB}$ 是与粒子的内禀宇称和场定义相关的纯相因子（通常取为 $\pm 1$ 或 $\pm i$）

\vspace{2ex}
\begin{center}
    \large\text{标准$boost$获得动量}
\end{center}
\vspace{2ex}
已经确定了静止参考系下的系数 $V_{ab}(0, \sigma)$ 正比于 C-G 系数。那么如何获得具有任意有限动量 $\boldsymbol{p}$ 的物理态系数 $u_{ab}(\boldsymbol{p}, \sigma)$？ 

采取的办法是做一个$boost$获得有限动，由于 $\mathscr{D}(\Lambda)$ 是洛伦兹群的表示矩阵，满足 $\mathscr{D}(L(p)) \mathscr{D}(L^{-1}(p)) = \mathbb{I}$。我们在方程\eqref{eq:8.38a} 两边同时左乘 $\mathscr{D}(L(p))$，以解出动量为 $\boldsymbol{p}$ 时的场系数：
\begin{equation}
    \sum_{\bar{\sigma}}v_{ab}(\boldsymbol{p},\bar{\sigma})\delta_{\sigma\bar{\sigma}}
        =\sqrt{\frac{m}{p^0}}\sum_{\bar{a}\bar{b}}\mathscr{D}_{ab;\bar{a}\bar{b}}(L(p))v_{\bar{a}\bar{b}}(0,\sigma)
    \label{eq:u_boosted}
\end{equation}
物理图像到这里变得无比清晰：\color{blue}要构造一个动量为 $\boldsymbol{p}$、自旋投影为 $\sigma$ 的系数函数，我们只需要先在静止系用 C-G 系数把它“拼装”出来，然后再用一个标准的洛伦兹 Boost 矩阵 $\mathscr{D}(L(p))$ 把整个张量“加速”到动量 $p$ 即可。\normalcolor

为了让这个公式具备实际的可计算性，需要写出 $\mathscr{D}(L(p))$ 在 $(A,B)$ 表示下的具体指数形式。设 Boost 的快度矢量为 $\boldsymbol{\phi} = \hat{\boldsymbol{p}} \text{arctanh}(|\boldsymbol{p}|/p^0)$\footnote{
为了计算有限动量态的旋量系数，引入了“Rapidity” $\phi$ 作为核心参数，其定义为：
\begin{align*}
    \cosh\phi &= \frac{\sqrt{\boldsymbol{k}^2+m^2}}{m} \\
    \sinh\phi&= \frac{|\boldsymbol{k}|}{m}
\end{align*}
这种参量化的最大物理优势在于，同向的 Boost 变换具有简单的加法性质：$L(\bar{\phi})L(\phi) = L(\bar{\phi}+\phi)$。
}对应的生成元为 $\boldsymbol{K}$。在 $(A,B)$ 表示中，生成元满足：
\begin{equation}
    \boldsymbol{J} = \mathbf{A} + \mathbf{B}, \quad \boldsymbol{k} = -i\mathbf{A} + i\mathbf{B}
\end{equation}
所以，有限 Boost 操作的表示矩阵可以彻底分解为左右两部分的张量积：
\begin{align}
    \mathscr{D}\bigl(L(q)\bigr) &= \exp\left( -i \boldsymbol{\phi} \cdot \boldsymbol{k} \right) \nonumber \\
    &= \exp\left( -i \boldsymbol{\phi} \cdot (-i\mathbf{A} + i\mathbf{B}) \right) \nonumber \\
    &= \exp\left( -\boldsymbol{\phi} \cdot \mathbf{A} \right) \otimes \exp\left( +\boldsymbol{\phi} \cdot \mathbf{B} \right)
\end{align}
将此矩阵代入 \eqref{eq:u_boosted}，，结合静止系下的 C-G 系数结果，最终得到了场论中\textbf{最一般的动力学系数展开式}：
\begin{equation}
   \sum_{\bar{\sigma}} v_{ab}(\boldsymbol{p}, \bar{\sigma}) \delta_{\sigma\bar{\sigma}}= \frac{1}{\sqrt{2p^0}} \sum_{\bar{a}\bar{b}} \left[ e^{-\boldsymbol{\phi} \cdot \mathbf{A}} \right]_{a\bar{a}} \left[ e^{+\boldsymbol{\phi} \cdot \mathbf{B}} \right]_{b\bar{b}} \langle A,a; B,b \,|\, j,\bar{\sigma} \rangle
\end{equation}
而反粒子对应的系数的是：
\begin{equation}
    u_{ab}(\boldsymbol{p}, \sigma)=(-)^{j+\sigma}v_{ab}(\boldsymbol{p}, -\sigma)
\end{equation}

\vspace{2ex}
\begin{center}
    \large\text{微观因果性和统计性质}
\end{center}
\vspace{2ex}
微观因果性的要求是：对于任意空间分离的事件 $x$ 和 $y$，场算符 $\Psi_{ab}(x)$ 和 $\widetilde{\Psi}_{cd}(y)$ 的（反）对易子必须为零：
\begin{equation}
    [\Psi_{ab}(x), \widetilde{\Psi}_{cd}^{\dagger}(y)]_{\mp} = 0 \quad \text{for} \quad (x-y)^2 < 0
\end{equation}  
考虑描述\textbf{同一种}自旋为$j$、质量为$m$的粒子的两个因果场：$\Psi_{ab}(x)$
按表示$(A,B)$变换，$\widetilde{\Psi}_{cd}(x)$按表示$(C,D)$变换。
它们的一般形式分别为
\begin{align}
  \Psi_{ab}(x) &= \frac{1}{(2\pi)^{3/2}}\sum_{\sigma}\int d^{3}p\,
  \bigl\{\xi_{ab}\,u_{ab}(\boldsymbol{p},\sigma)\,a^{c\dagger}(\boldsymbol{p},\sigma)\,e^{-ip\cdot x}
  + \lambda_{ab}\,v_{ab}(\boldsymbol{p},\sigma)\,a(\boldsymbol{p},\sigma)\,e^{ip\cdot x}\bigr\}\;,
  \label{eq:field-AB}\\[4pt]
  \widetilde{\Psi}_{cd}(x) &= \frac{1}{(2\pi)^{3/2}}\sum_{\sigma}\int d^{3}p\,
  \bigl\{\xi_{cd}\,u_{cd}(\boldsymbol{p},\sigma)\,a^{c\dagger}(\boldsymbol{p},\sigma)\,e^{-ip\cdot x}
  + \lambda_{cd}\,v_{cd}(\boldsymbol{p},\sigma)\,a(\boldsymbol{p},\sigma)\,e^{ip\cdot x}\bigr\}\;.
  \label{eq:field-CD}
\end{align}

利用基本（反）对易关系
\[
  [a(\boldsymbol{p},\sigma),\,a^{\dagger}(\boldsymbol{q},\sigma')]_{\mp}
  = \delta^{3}(\boldsymbol{p}-\boldsymbol{q})\,\delta_{\sigma\sigma'}\;,\qquad
  [a^{c\dagger}(\boldsymbol{p},\sigma),\,a^{c}(\boldsymbol{q},\sigma')]_{\mp}
  = \mp\,\delta^{3}(\boldsymbol{p}-\boldsymbol{q})\,\delta_{\sigma\sigma'}\;,
\]
直接计算得到
\begin{equation}\label{eq:comm-general}
  [\Psi_{ab}(x),\,\widetilde{\Psi}_{cd}^{\dagger}(y)]_{\mp}
  = \frac{1}{(2\pi)^{3}}\int\frac{d^{3}p}{2p^{0}}\,
    \pi_{ab,cd}(\boldsymbol{p})\,
    \Bigl[\lambda_{ab}\lambda_{cd}^{*}\,e^{ip\cdot(x-y)}
      \;\mp\; \xi_{ab}\xi_{cd}^{*}\,e^{-ip\cdot(x-y)}\Bigr]
\end{equation}
函数$\pi_{ab,cd}(\boldsymbol{p})$定义为
\begin{equation}\label{eq:pi-def}
  \frac{\pi_{ab,cd}(\boldsymbol{p})}{2p^{0}}
  \;\equiv\; \sum_{\sigma}u_{ab}(\boldsymbol{p},\sigma)\,
    u_{cd}^{*}(\boldsymbol{p},\sigma)
  \;=\; \sum_{\sigma}v_{ab}(\boldsymbol{p},\sigma)\,
    v_{cd}^{*}(\boldsymbol{p},\sigma)\;.
\end{equation}
第二个等号利用了$u_{ab}(\boldsymbol{p},\sigma)=(-)^{j+\sigma}\,v_{ab}(\boldsymbol{p},-\sigma)$
（对$v$同理），对$\sigma$重标记即得。
\vspace{2ex}
\\
$\pi_{ab,cd}(\boldsymbol{p})$的多项式结构;为看清$\pi$的宇称性质，取$\boldsymbol{p}$沿$3$-方向：
$\boldsymbol{p}=(0,0,|\boldsymbol{p}|)$。此时标准Lorentz boost
$L(p)$退化为沿$3$-轴的纯boost，快度$\phi$满足
\[
  \cosh\phi = \frac{p^{0}}{m}\;,\qquad
  \sinh\phi = \frac{|\boldsymbol{p}|}{m}\;,\qquad
  e^{\phi} = \frac{p^{0}+p^{3}}{m}\;.
\]
在$(A,B)$表示中，$D^{(A,B)}(L(p))=e^{-\phi A_{3}}\otimes e^{+\phi B_{3}}$
是完全对角的：
\[
  [e^{-\phi A_{3}}]_{a\bar{a}} = e^{-a\phi}\,\delta_{a\bar{a}}\;,\qquad
  [e^{+\phi B_{3}}]_{b\bar{b}} = e^{+b\phi}\,\delta_{b\bar{b}}\;.
\]
因此$u_{ab}$中的boost矩阵只贡献因子$e^{(-a+b)\phi}$；
对$(C,D)$表示的$u_{cd}$类似，贡献$e^{(-c+d)\phi}$。
Clebsch--Gordan系数要求$\sigma = a+b = c+d$，
因此指数中的组合化简为
\[
  -a+b-c+d = -a+(\sigma-a)-(\sigma-d)+d = -2a+2d\;.
\]
代入(\ref{eq:pi-def})得到
\begin{equation}\label{eq:pi-explicit}
  \pi_{ab,cd}(\boldsymbol{p})
  = \sum_{\sigma}\langle A,a;\,B,b\,|\,j,\sigma\rangle
    \,\langle C,c;\,D,d\,|\,j,\sigma\rangle
    \;\cdot\;\left(\frac{p^{0}+p^{3}}{m}\right)^{\!-2a+2d}\;.
\end{equation}
当$d\geqslant a$时，$-2a+2d$是非负整数，
$(p^{0}+p^{3})^{-2a+2d}$展开后是$p^{0}$和$p^{3}$的多项式；
当$a > d$时，利用$e^{-\phi}=(p^{0}-p^{3})/m$同样给出多项式。
在质量壳$p^{0}=\sqrt{\boldsymbol{p}^{2}+m^{2}}$上，这些幂次至多对
$p^{0}$是线性的（因为$(p^{0})^{2}=\boldsymbol{p}^{2}+m^{2}$
可消去更高次幂），故$\pi$是$\boldsymbol{p}$和$p^{0}$的多项式在质量壳上的取值。
由Lorentz协变性，上述结论对任意方向的$\boldsymbol{p}$均成立。
\vspace{2ex}\\
反射条件：在$p^{\mu}\to -p^{\mu}$（即$p^{0}\to -p^{0}$，$\boldsymbol{p}\to -\boldsymbol{p}$）下，
(\ref{eq:pi-explicit})中的因子变为
\[
  \left(\frac{p^{0}+p^{3}}{m}\right)^{-2a+2d}
  \;\longrightarrow\;
  \left(\frac{-(p^{0}+p^{3})}{m}\right)^{-2a+2d}
  = (-)^{-2a+2d}\left(\frac{p^{0}+p^{3}}{m}\right)^{-2a+2d}\;.
\]
由于$a$的取值与$A$差一个整数，$(-)^{-2a}=(-)^{2A}$；
类似地$(-)^{2d}=(-)^{2D}$。因此
\[
  (-)^{-2a+2d} = (-)^{2A+2D}\;,
\]
该相因子对所有允许的$a,d$取值均相同。
设$\pi$的多项式记为$\mathscr{P}(\boldsymbol{p},p^{0})$，则反射条件为
\begin{equation}\label{eq:reflect}
  \mathscr{P}(-\boldsymbol{p},-p^{0})
  = (-)^{2A+2D}\,\mathscr{P}(\boldsymbol{p},p^{0})\;.
\end{equation}
任何$\boldsymbol{p}$和$p^{0}=\sqrt{\boldsymbol{p}^{2}+m^{2}}$的多项式
至多对$p^{0}$线性（因$(p^{0})^{2}=\boldsymbol{p}^{2}+m^{2}$），
故可分解为
\begin{equation}\label{eq:PQ-decomp}
  \pi_{ab,cd}(\boldsymbol{p})
  = P_{ab,cd}(\boldsymbol{p})
  + 2p^{0}\,Q_{ab,cd}(\boldsymbol{p})\;,
\end{equation}
其中$P$和$Q$仅是$\boldsymbol{p}$的多项式。
将反射条件(\ref{eq:reflect})代入，注意
$p^{0}\to -p^{0}$时
$\sqrt{\boldsymbol{p}^{2}+m^{2}}$变号而$\boldsymbol{p}\to -\boldsymbol{p}$下不变，
对比含$p^{0}$与不含$p^{0}$的部分，得
\begin{align}
  P_{ab,cd}(-\boldsymbol{p})
  &= (-)^{2A+2D}\,P_{ab,cd}(\boldsymbol{p})\;,
  \label{eq:P-reflect}\\[2pt]
  Q_{ab,cd}(-\boldsymbol{p})
  &= -(-)^{2A+2D}\,Q_{ab,cd}(\boldsymbol{p})\;.
  \label{eq:Q-reflect}
\end{align}
回到对易条件，因果性要求：对类空间隔$(x-y)^{2}<0$，
$[\Psi_{ab}(x),\widetilde{\Psi}_{cd}^{\dagger}(y)]_{\mp}=0$。
对于类空间隔，总可以选择一个参考系使
$x^{0}=y^{0}$而$\boldsymbol{x}\neq\boldsymbol{y}$%
\footnote{场的（反）对易子在Lorentz变换下协变变换：
若它在某个参考系中为零，
则在任何参考系中都为零。选择等时参考系的技术好处在于，
指数中$p^{0}$的时间部分消失，
使得$P$项和$Q$项产生完全不同的空间结构，
从而可以分离出约束条件。}。
记$\mathbf{r}=\boldsymbol{x}-\boldsymbol{y}$，则
$e^{\pm ip\cdot(x-y)}=e^{\pm i\boldsymbol{p}\cdot\mathbf{r}}$。
代入(\ref{eq:comm-general})：
\[
  [\Psi_{ab},\widetilde{\Psi}_{cd}^{\dagger}]_{\mp}\Big|_{x^{0}=y^{0}}
  = \frac{1}{(2\pi)^{3}}\int\frac{d^{3}p}{2p^{0}}\,\pi(\boldsymbol{p})\,
    \bigl\{\lambda_{ab}\lambda_{cd}^{*}\,e^{i\boldsymbol{p}\cdot\mathbf{r}}
    \;\mp\; \xi_{ab}\xi_{cd}^{*}\,e^{-i\boldsymbol{p}\cdot\mathbf{r}}\bigr\}\;.
\]
对第二项做变量替换$\boldsymbol{p}\to -\boldsymbol{p}$
（积分测度$d^{3}p$不变，$p^{0}$不变，
$e^{-i\boldsymbol{p}\cdot\mathbf{r}}\to e^{i\boldsymbol{p}\cdot\mathbf{r}}$，
$\pi(\boldsymbol{p})\to \pi(-\boldsymbol{p})$），两项合并得
\[
  = \frac{1}{(2\pi)^{3}}\int\frac{d^{3}p}{2p^{0}}\,
    \bigl\{\lambda_{ab}\lambda_{cd}^{*}\,\pi(\boldsymbol{p})
    \;\mp\; \xi_{ab}\xi_{cd}^{*}\,\pi(-\boldsymbol{p})\bigr\}
    \,e^{i\boldsymbol{p}\cdot\mathbf{r}}\;.
\]
将(\ref{eq:PQ-decomp})和反射性质
$\pi(-\boldsymbol{p})=(-)^{2A+2D}\bigl[P(\boldsymbol{p})-2p^{0}\,Q(\boldsymbol{p})\bigr]$
代入大括号：
\begin{align}
  &\lambda_{ab}\lambda_{cd}^{*}\bigl[P+2p^{0}Q\bigr]
   \;\mp\; \xi_{ab}\xi_{cd}^{*}\,(-)^{2A+2D}
       \bigl[P-2p^{0}Q\bigr] \notag\\[3pt]
  &\;= \bigl[\lambda_{ab}\lambda_{cd}^{*}
         \mp (-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]\,P
    \;+\; \bigl[\lambda_{ab}\lambda_{cd}^{*}
         \pm (-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]
           \cdot 2p^{0}\,Q\;.
  \label{eq:PQ-split}
\end{align}

现在分别考察$P$项和$Q$项的空间积分。
定义
\[
  \Delta_{+}(\mathbf{r},0)
  \;\equiv\; \frac{1}{(2\pi)^{3}}\int\frac{d^{3}p}{2p^{0}}\,
    e^{i\boldsymbol{p}\cdot\mathbf{r}}\;.
\]

\noindent\textbf{$P$项}：
由于$P(\boldsymbol{p})$是$\boldsymbol{p}$的多项式，
可用$(-i\nabla)e^{i\boldsymbol{p}\cdot\mathbf{r}}=\boldsymbol{p}\,e^{i\boldsymbol{p}\cdot\mathbf{r}}$
将$P(\boldsymbol{p})$替换为微分算符$P(-i\nabla)$提到积分号外，剩下
\[
  \bigl[\lambda_{ab}\lambda_{cd}^{*}
    \mp (-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]\;
    P(-i\nabla)\,\Delta_{+}(\mathbf{r},0)\;.
\]
$\Delta_{+}(\mathbf{r},0)$是一个在$\mathbf{r}\neq\mathbf{0}$
处处处非零的光滑函数（其各阶导数亦一般非零）。

\medskip
\noindent\textbf{$Q$项}：
$2p^{0}$与分母的$2p^{0}$精确抵消，剩下
\[
  \bigl[\lambda_{ab}\lambda_{cd}^{*}
    \pm (-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]\;
  \int\frac{d^{3}p}{(2\pi)^{3}}\,Q(\boldsymbol{p})\,e^{i\boldsymbol{p}\cdot\mathbf{r}}
  \;=\;
  \bigl[\cdots\bigr]\;Q(-i\nabla)\,\delta^{3}(\mathbf{r})\;.
\]
此项在$\mathbf{r}\neq\mathbf{0}$时恒为零。

\vspace{2ex}
\medskip
综合两项，等时对易子为
\begin{align}
  [\Psi_{ab},\widetilde{\Psi}_{cd}^{\dagger}]_{\mp}
  \Big|_{x^{0}=y^{0}}
  &= \bigl[\lambda_{ab}\lambda_{cd}^{*}
     \mp(-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]\;
     P(-i\nabla)\,\Delta_{+}(\mathbf{r},0)
     \notag\\[3pt]
  &\quad
   + \bigl[\lambda_{ab}\lambda_{cd}^{*}
     \pm(-)^{2A+2D}\,\xi_{ab}\xi_{cd}^{*}\bigr]\;
     Q(-i\nabla)\,\delta^{3}(\mathbf{r})\;.
  \label{eq:equal-time}
\end{align}
因果性要求此式在$\mathbf{r}\neq\mathbf{0}$时为零。
$\delta^{3}$项自动消失；$\Delta_{+}$项一般非零，
唯一使其消失的办法是令系数为零：
\begin{equation}\label{eq:causality-cond}
  \xi_{ab}\xi_{cd}^{*}
  = \pm(-)^{2A+2D}\;\lambda_{ab}\lambda_{cd}^{*}
\end{equation}
先考察$\Psi$和$\widetilde{\Psi}$\textbf{相同}的特殊情形，即
$C=A$，$D=B$，$\xi_{cd}=\xi_{ab}$，$\lambda_{cd}=\lambda_{ab}$。
(\ref{eq:causality-cond})化为
\begin{equation}\label{eq:same-field}
  |\xi_{ab}|^{2} = \pm(-)^{2A+2B}\,|\lambda_{ab}|^{2}\;.
\end{equation}
由于两边均为正实数，必须
\begin{equation}\label{eq:spin-stat}
  \pm(-)^{2A+2B} = +1
\end{equation}
因$2A+2B$与$2j$（自旋）同奇偶（二者之差为偶数），
(\ref{eq:spin-stat})意味着：
\begin{itemize}
  \item $2j$为偶（整数自旋）：取上号（$+$），
        对应\textbf{对易子}——玻色子；
  \item $2j$为奇（半整数自旋）：取下号（$-$），
        对应\textbf{反对易子}——费米子。
\end{itemize}
同时还有
\begin{equation}\label{eq:mod-equal}
  |\xi_{ab}|^{2}=|\lambda_{ab}|^{2}\;.
\end{equation}
这就是\textbf{自旋与统计之间的普遍关系}。

\vspace{2ex}
\medskip
回到$\Psi$和$\widetilde{\Psi}$不同的一般情形。
(\ref{eq:causality-cond})对$\Psi$自身
（$\widetilde{\Psi}=\Psi$）及$\widetilde{\Psi}$自身
（$\Psi=\widetilde{\Psi}$）分别成立，给出
$|\xi_{ab}|=|\lambda_{ab}|$和$|\xi_{cd}|=|\lambda_{cd}|$。
在(\ref{eq:causality-cond})中同除以$|\xi_{cd}|^{2}=|\lambda_{cd}|^{2}$，
左边利用$\xi_{cd}^{*}/|\xi_{cd}|^{2}=1/\xi_{cd}$，
右边利用$\lambda_{cd}^{*}/|\lambda_{cd}|^{2}=1/\lambda_{cd}$，得
\[
  \frac{\xi_{ab}}{\xi_{cd}}
  = \pm(-)^{2A+2D}\;\frac{\lambda_{ab}}{\lambda_{cd}}\;.
\]
将自旋—统计关系$\pm 1=(-)^{2A+2B}$代入：
\begin{align}
  \frac{\xi_{ab}}{\xi_{cd}}
  &= (-)^{2A+2B}\cdot(-)^{2A+2D}\;
    \frac{\lambda_{ab}}{\lambda_{cd}}
  = (-)^{4A+2B+2D}\;\frac{\lambda_{ab}}{\lambda_{cd}}\;.\notag
\end{align}
因$4A$为偶整数，$(-)^{4A}=1$，指数简化为$2B+2D$：
\[
  \frac{\xi_{ab}}{\xi_{cd}}
  = (-)^{2B+2D}\;\frac{\lambda_{ab}}{\lambda_{cd}}\;.
\]
改写为
\[
  \frac{\xi_{ab}}{(-)^{2B}\,\lambda_{ab}}
  = \frac{\xi_{cd}}{(-)^{2D}\,\lambda_{cd}}\;.
\]
左边只含场$\Psi$的参数$B$，
右边只含场$\widetilde{\Psi}$的参数$D$，
但两者相等——说明这个比值与场的选择无关，
仅取决于粒子本身。常数为$c$实际上总可以吸收进场算符中，不过在这里我们取$c=1$：
\begin{equation}\label{eq:xi-lambda}
  \xi_{ab} = (-)^{2B}\,c\,\lambda_{ab}\;,\qquad |c|=1\;.
\end{equation}
通过对算符$a(\boldsymbol{p},\sigma)$和$a^{c\dagger}(\boldsymbol{p},\sigma)$
的相对相位重定义，可令$c=1$；
再通过场的总体尺度重定义消除$\lambda_{ab}$。
最终，对于描述给定粒子的$(A,B)$场，唯一的因果场
（直到总体尺度）为
\begin{equation}\label{eq:final-field}
  \boxed{
  \Psi_{ab}(x)
  = \frac{1}{(2\pi)^{3/2}}\sum_{\sigma}\int d^{3}p\,
    \Bigl[(-)^{2B}\,u_{ab}(\boldsymbol{p},\sigma)\,
      a^{c\dagger}(\boldsymbol{p},\sigma)\,e^{-ip\cdot x}
    + v_{ab}(\boldsymbol{p},\sigma)\,
      a(\boldsymbol{p},\sigma)\,e^{ip\cdot x}\Bigr]
  }
\end{equation}
\begin{itemize}
  \item 产生部分$\bigl(a^{c\dagger}\,e^{-ipx}\bigr)$的系数
        由$(-)^{2B}$确定，湮灭部分$\bigl(a\,e^{ipx}\bigr)$
        的系数归一化为$1$。
  \item 整数自旋场满足对易关系（玻色子），
        半整数自旋场满足反对易关系（费米子）。
  \item 对于同一粒子的不同表示$(C,D)$的场
        $\widetilde{\Psi}_{cd}$，产生部分的系数换为$(-)^{2D}$，
        常数$c$相同。
\end{itemize}
