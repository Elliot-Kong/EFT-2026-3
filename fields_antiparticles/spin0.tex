\subsection{因果标量场 }
一般形式的场是：
\begin{equation}
    \Psi_{ab}(x) =(\frac{1}{2\pi})^\frac{3}{2}\sum_{\sigma}\int d^3p \, \left\{ (-)^{2B}(-)^{j+\sigma}v_{ab}(\boldsymbol{q},\sigma)a^{c\dagger}(\boldsymbol{q},\sigma) e^{-ipx} + v_{ab}(\boldsymbol{q},\sigma) a(\boldsymbol{q},\sigma)e^{ipx} \right\}
\end{equation}
上述我们知道静止系系数$v(0)$ 是正比于Clebsch-Gordan  系数 
\begin{equation}
    v_{ab}(0, \sigma) = \frac{1}{\sqrt{2m}} \langle A, a; B, b | j, \sigma \rangle
    \label{eq:u0_general}
\end{equation}
通过 Boost 获得有限动量系数 $u(\boldsymbol{q})$ 和 $v(\boldsymbol{q})$的一般公式是：
\begin{equation}
    v_{ab}(\boldsymbol{p}, \sigma) = \frac{1}{\sqrt{2p^0}} \sum_{\bar{a},\bar{b}} \left[e^{-\boldsymbol{\phi}\cdot\mathbf{J}^{(A)}}\right]_{a\bar{a}} \left[e^{+\boldsymbol{\phi}\cdot\mathbf{J}^{(B)}}\right]_{b\bar{b}} \langle A,\bar{a}; B,\bar{b} | j, \sigma \rangle
    \label{eq:u_finite_general}
\end{equation}
其中$\boldsymbol{\phi} = \hat{\boldsymbol{p}} \text{arctanh}(|\boldsymbol{p}|/p^0)$

\topictitle{$(A, B) = (0, 0)$}

对于标量场，$(A, B) = (0, 0)$，此时：
\begin{itemize}
    \item 角动量生成元全为零：$\mathbf{J}^{(A)} = \mathbf{J}^{(B)} = 0$。
    \item 表示指标只有一个值：$a = 0$，$b = 0$。
    \item 自旋投影只有一个值：$\sigma = 0$。
    \item $j=0$根据自旋统计表明是波色子。
\end{itemize}
代入式 \eqref{eq:u0_general}，唯一的 C-G 系数为 $\langle 0, 0; 0, 0 | 0, 0 \rangle = 1$。因此静止系下的系数为：
\begin{equation}
    v(0, 0) = \frac{1}{\sqrt{2m}}
\end{equation}
\vspace{2ex}
对于标量场，$\mathbf{J}^{(A)} = 0$ 且 $\mathbf{J}^{(B)} = 0$。因此，指数演化矩阵完全退化为恒等映射：
\begin{equation}
    \exp(-\boldsymbol{\phi}\cdot\mathbf{0}) = 1, \quad \exp(+\boldsymbol{\phi}\cdot\mathbf{0}) = 1
\end{equation}
代入式 \eqref{eq:u_finite_general}，并利用 $\langle 0,0;0,0|0,0\rangle = 1$，有限动量系数极为简单：
\begin{equation}
    v(\boldsymbol{p}, 0) = \frac{1}{\sqrt{2p^0}}
\end{equation}


\paragraph{$\pi$函数与多项式结构.}
由定义
$\pi(\mathbf{p})/(2p^{0})
=\sum_{\sigma}|u(\mathbf{p},\sigma)|^{2}
=1/(2p^{0})$，
立即得到
\begin{equation}\label{eq:pi-scalar}
  \pi(\mathbf{p})=1\;.
\end{equation}
这是一个常数（零次多项式），自然满足分解
$\pi = P + 2p^{0}Q$中$P=1$，$Q=0$。
反射条件$P(-\mathbf{p})=(-)^{2A+2D}\,P(\mathbf{p})$
退化为$1=(-)^{0}\cdot 1$，平凡成立。
\paragraph{因果场的形式.}
由$(-)^{2B}=(-)^{0}=1$，
一般结果$\xi = (-)^{2B}\,c\,\lambda = c\,\lambda$
退化为$\xi=c\lambda$。
取$c=1$并归一化$\lambda=1$，
唯一的因果标量场为
\begin{equation}\label{eq:scalar-field}
  \boxed{
  \Psi(x)
  = \frac{1}{(2\pi)^{3/2}}\int\frac{d^{3}p}{\sqrt{2p^{0}}}
    \Bigl[a^{c\dagger}(\mathbf{p})\,e^{-ip\cdot x}
    + a(\mathbf{p})\,e^{ip\cdot x}\Bigr]
  }
\end{equation}
若粒子是自共轭的（实标量场，不携带守恒荷），
$a^{c\dagger}(\mathbf{p})=a^{\dagger}(\mathbf{p})$，
场化为
\begin{equation}\label{eq:real-scalar}
  \phi(x)
  = \frac{1}{(2\pi)^{3/2}}\int\frac{d^{3}p}{\sqrt{2p^{0}}}
    \Bigl[a^{\dagger}(\mathbf{p})\,e^{-ip\cdot x}
    + a(\mathbf{p})\,e^{ip\cdot x}\Bigr]\;.
\end{equation}

\paragraph{验证因果性.}
对实标量场直接计算对易子：
\begin{align}
  [\phi(x),\,\phi(y)]
  &= \frac{1}{(2\pi)^{3}}\int\frac{d^{3}p}{2p^{0}}\,
    \Bigl[e^{ip\cdot(x-y)} - e^{-ip\cdot(x-y)}\Bigr]
    \notag\\[3pt]
  &\equiv i\Delta(x-y)\;.
  \label{eq:scalar-commutator}
\end{align}
这正是$\pi=1$（$P=1$，$Q=0$）、
$|\xi|^{2}=|\lambda|^{2}=1$
代入一般公式(\ref{eq:comm-general})的直接结果。
函数$i\Delta(x-y)$是一个Lorentz不变量%
\footnote{
  因为$\Delta(x-y)$仅依赖于不变间隔
  $(x-y)^{2}=\eta_{\mu\nu}(x-y)^{\mu}(x-y)^{\nu}$。
}，
可以证明它在类空间隔$(x-y)^{2}<0$下恒为零。
最直观的验证方式如下：
\begin{itemize}
  \item 类空间隔下，总可以找到一个Lorentz变换
        将$x-y$反转为$y-x$%
        \footnote{
          精确地说：对于类空$x-y$，
          存在$\Lambda\in SO^{+}(1,3)$使得
          $\Lambda(x-y)=-(x-y)$。
          例如可以选取在等时面上的
          空间反射$\mathbf{x}-\mathbf{y}\to
          -(\mathbf{x}-\mathbf{y})$。
        }。
        由于$\Delta_{+}(x-y)$是Lorentz标量，
        在此变换下$\Delta_{+}(x-y)=\Delta_{+}(y-x)$；
  \item 因此$i\Delta(x-y)
        =\Delta_{+}(x-y)-\Delta_{+}(y-x)=0$。
\end{itemize}




\paragraph{复标量场.}
若粒子携带守恒荷$q$（如电荷），
则其反粒子不等于自身：$a^{c\dagger}\neq a^{\dagger}$。
此时一般场(\ref{eq:scalar-field})中$a^{c\dagger}$
产生反粒子，$a$湮灭粒子。
场$\Psi$和$\Psi^{\dagger}$分别减少和增加系统的荷$q$：
\[
  [Q,\Psi(x)]=-q\,\Psi(x)\;,\qquad
  [Q,\Psi^{\dagger}(x)]=+q\,\Psi^{\dagger}(x)\;.
\]
相互作用密度$\mathscr{H}(x)$中$\Psi$和$\Psi^{\dagger}$
必须成对出现以保证$[Q,\mathscr{H}]=0$。
因果律的验证完全平行于实标量情形，
结论不变：自旋$0$携荷粒子仍是玻色子，
场满足对易关系且满足Klein--Gordon方程。
