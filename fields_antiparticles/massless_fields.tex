\section{零质量粒子因果场\(\dagger\)}
开始构造最一般零质量粒子场前，我们先对零质量粒子的小群 $ISO(2)$ 进行分析，看看它的生成元在 $(A,B)$ 表示中的分解情况。
\begin{center}\vspace{2ex}
\textbf{$ISO(2)$ 生成元在 $(A,B)$ 表示中的分解}
\vspace{2ex}\end{center}
$ISO(2)$ 小群（标准动量 $k^\mu=(\kappa,0,0,\kappa)$）的三个生成元为

\begin{equation}
    J_3= A_3 + B_3\,,\qquad \Pi_1 = K_1 - J_2\,,\qquad \Pi_2 = K_2 + J_1
\end{equation}
在前文的 Wigner 小群分析中，为了排除违背物理现实的“连续自旋”，我们必须强制要求物理态在“平移”生成元下特征值为零，也就是要求方程 
\begin{subequations}
    \begin{align}
        \Pi_1 \, \Ket{k,\lambda} &= 0 \\
        \Pi_2 \, \Ket{k,\lambda} &= 0
    \end{align}
\end{subequations}
不过，这两个式子解不出有效的信息；尝试构造 $\Pi_1 \pm i\Pi_2$ 的线性组合，看看能否得到更有用的约束。
在 $(A,B)$ 表示中，利用 Lorentz 群生成元的分解 $\mathbf{J} = \mathbf{A} + \mathbf{B}$，$\boldsymbol{k} = -i\mathbf{A} + i\mathbf{B}$，以及升降算符 $A_\pm = A_1 \pm iA_2$ 与 $B_\pm = B_1 \pm iB_2$
\begin{itemize}
    \item 计算 $\Pi_1 + i\Pi_2$ 的详细展开：
    \begin{align}
        \Pi_1 + i\Pi_2 &= (K_1 - J_2) + i(K_2 + J_1) \notag \\
        &= (K_1 + iK_2) + (iJ_1 - J_2) \notag \\
        \text{（代入生成元分解）} &= \bigl[ (-iA_1 + iB_1) + i(-iA_2 + iB_2) \bigr] + \bigl[ i(A_1 + B_1) - (A_2 + B_2) \bigr] \notag \\
        &= \bigl( -iA_1 + A_2 + iB_1 - B_2 \bigr) + \bigl( iA_1 - A_2 + iB_1 - B_2 \bigr) \notag \\
        \text{（重新分组）} &= \underbrace{\bigl( -iA_1 + A_2 + iA_1 - A_2 \bigr)}_{=\, 0 \text{ （}A\text{ 部分完全抵消）}} + \underbrace{\bigl( iB_1 - B_2 + iB_1 - B_2 \bigr)}_{=\, 2iB_1 - 2B_2} \notag \\
        &= 2i \bigl( B_1 + iB_2 \bigr) \notag \\
        &= 2i B_+
    \end{align}
    \item 计算 $\Pi_1 - i\Pi_2$ 的详细展开：   
    \begin{align}
        \Pi_1 - i\Pi_2 &= (K_1 - J_2) - i(K_2 + J_1) \notag \\
        &= (K_1 - iK_2) + (-iJ_1 - J_2) \notag \\
        \text{（代入生成元分解）} &= \bigl[ (-iA_1 + iB_1) - i(-iA_2 + iB_2) \bigr] + \bigl[ -i(A_1 + B_1) - (A_2 + B_2) \bigr] \notag \\
        &= \bigl( -iA_1 - A_2 + iB_1 + B_2 \bigr) + \bigl( -iA_1 - A_2 - iB_1 - B_2 \bigr) \notag \\
        \text{（重新分组）} &= \underbrace{\bigl( -iA_1 - A_2 - iA_1 - A_2 \bigr)}_{=\, -2iA_1 - 2A_2} + \underbrace{\bigl( iB_1 + B_2 - iB_1 - B_2 \bigr)}_{=\, 0 \text{ （}B\text{ 部分完全抵消）}} \notag \\
        &= -2i \bigl( A_1 - iA_2 \bigr) \notag \\
        &= -2i A_-
\end{align}
\end{itemize}
得出 $ISO(2)$ 的“平移”生成元在 $(A,B)$ 表示中完全分解为：
\begin{equation}
 \ \Pi_1 + i\Pi_2 = 2i B_+, \qquad \Pi_1 - i\Pi_2 = -2i A_- \ 
\end{equation}
为了排除物理上未观测到的“连续自旋”态，我们强制要求物理态在“平移”生成元下的特征值为零。也就是：
\begin{equation}
    \Pi_1 \, \Ket{k,\lambda} = 0, \qquad \Pi_2 \,\Ket{k,\lambda} = 0
\end{equation}
这等等价于 $(\Pi_1 \pm i\Pi_2) \ket{k,\lambda} = 0$。代入我们在上一小节得到的代数分解式 $\Pi_1 + i\Pi_2 = 2i B_+$ 与 $\Pi_1 - i\Pi_2 = -2i A_-$，得到了对交织方程的极端强约束：
\begin{subequations}
    \begin{align}
    B_+ \, \ket{k,a,b} &= 0 \quad \implies \quad u_{ab} \text{ 必须是 } B \text{ 空间的最高权态，即 :} b = +B \label{eq:B_plus_zero} \\
    A_- \, \ket{k,a,b} &= 0 \quad \implies \quad u_{ab} \text{ 必须是 } A \text{ 空间的最低权态，即 :} a = -A \label{eq:A_minus_zero}
\end{align}
\end{subequations}
也就是说，此刻的本征态 $\ket{k,a,b}$ 中的 $a$ 和 $b$ 已经被完全固定为 $a = -A$ 和 $b = +B$。现在只剩下最后一个生成元 $J_3 = A_3 + B_3$ 没有被约束了；所以有本征关系：
\begin{equation}
    J_3 \, \ket{k,-A,+B } = (A_3 + B_3) \, \ket{k,-A,+B} = (-A + B) \, \ket{k,-A,+B}
\end{equation}
这就得出了由 $(A, B)$ 场湮灭的无质量粒子的确定螺旋度：
\begin{equation}\label{helcity_o}
    \lambda = B - A
\end{equation}
\hrule\vspace{3ex}
非物理带来的约束问题已经讨论完毕，现在开始构造最一般的场，一般形式场写为：\begin{equation*}
    \Psi_\ell(x) =(\frac{1}{2\pi})^\frac{3}{2}\sum_{\sigma}\int d^3k \, \left\{\xi_\ell u_\ell(\boldsymbol{k},\sigma)a^{c\dagger}(\boldsymbol{k},\sigma) e^{-ikx} + \lambda_\ell v_\ell(\boldsymbol{k},\sigma) a(\boldsymbol{k},\sigma)e^{ikx} \right\}
\end{equation*}
若理论本身无荷，就有$a^{\dagger}(\boldsymbol{k},\sigma)=a^{c\dagger}(\boldsymbol{k},\sigma)=a^{\dagger}(\boldsymbol{k},\sigma)$；系数$u_\ell(\boldsymbol{k},\sigma)$和$v_\ell(\boldsymbol{k},\sigma)$的变换是
\begin{equation*}
    \begin{aligned}
            &\sum_{\bar{\sigma}}
            u_{\ell}(\boldsymbol{k},\bar{\sigma})
            \sqrt{\frac{k^0}{(\Lambda k)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n}
            \!\left\{W^{-1}(\Lambda,k)\right\} =
            \sum_{\bar{\ell}}
            \mathscr{D}_{\ell\bar{\ell}}(\Lambda^{-1})
            u_{\bar{\ell}}(\Lambda \boldsymbol{k},\sigma)\\
            &\sum_{\bar{\sigma}}
            v_{\ell}(\boldsymbol{k},\bar{\sigma})
            \sqrt{\frac{k^0}{(\Lambda k)^0}} 
            D_{\sigma \bar{\sigma}}^{j_n*}
            \!\left\{W^{-1}(\Lambda,k)\right\} =
            \sum_{\bar{\ell}}
            \mathscr{D}_{\ell\bar{\ell}}(\Lambda^{-1})
            v_{\bar{\ell}}(\Lambda \boldsymbol{k},\sigma)
            \end{aligned}
\end{equation*}
通过 $SU(2)_L \times SU(2)_R$ 代数构建Lorentz群的有限维表示后。指标$\ell$在这里被替换为一对指标$a$、$b$；以及零质量对应Helcity 所以替换$\sigma$为$\lambda$,所以一般形式场可以写为：
\begin{equation}
    \Psi_{ab}(x) =(\frac{1}{2\pi})^\frac{3}{2}\sum_{\lambda}\int d^3k \, \left\{\xi_{ab} u_{ab}(\boldsymbol{k},\lambda)a^{c\dagger}(\boldsymbol{k},\lambda) e^{-ikx} + \lambda_{ab} v_{ab}(\boldsymbol{k},\lambda) a(\boldsymbol{k},\lambda)e^{ikx  } \right\}
\end{equation}
以及系数函数的选择变为：
\begin{subequations}
    \begin{align}
            &\sum_{\bar{\lambda}}
            u_{ab}(\boldsymbol{k},\bar{\lambda})
            \sqrt{\frac{k^0}{(\Lambda k)^0}} 
            D_{\lambda \bar{\lambda}}^{j_n}
            \!\left\{W^{-1}(\Lambda,k)\right\} =
            \sum_{\bar{a}\bar{b}}
            \mathscr{D}_{{ab};\bar{a}\bar{{b}}}(\Lambda^{-1})
            u_{\bar{a}\bar{b}}(\Lambda \boldsymbol{k},\lambda)
            \\
             &\sum_{\bar{\lambda}}
            v_{ab}(\boldsymbol{k},\bar{\lambda})
            \sqrt{\frac{k^0}{(\Lambda k)^0}} 
            D_{\lambda \bar{\lambda}}^{j_n*}
            \!\left\{W^{-1}(\Lambda,k)\right\} =
            \sum_{\bar{a}\bar{b}}
            \mathscr{D}_{{ab};\bar{a}\bar{{b}}}(\Lambda^{-1})
            v_{\bar{a}\bar{b}}(\Lambda \boldsymbol{k},\lambda)
            \end{align}
\end{subequations}
首先确定系数的选择；考虑剩下的两组Poincar\'e变换，依次考察$\operatorname{Boost}$和$\operatorname{Rotation}$；
\vspace{2ex}
\begin{center}
    \large\textbf{boost}  
\end{center}
\vspace{2ex}
\begin{itemize}
    \item 令 $k^\mu = (|\boldsymbol{k}|,\boldsymbol{k})$为任意动量系;
    $p^\mu=(p,0,0,p)$为标准动量系；
    \item 由于零质量粒子，不存在所谓静止系；我们知道
    $$p^\mu = (p,0,0,p)\overset{\mathscr{L}(k)}{\longrightarrow }k^\mu(|\boldsymbol{k}|,\boldsymbol{k})$$
    所以令$\Lambda=\mathscr{L}^{-1}(k)$是 $p^\mu = (p,0,0,p)\to k^\mu $的 标准Lorentz变换,表示为：
    $$k^\mu(|\boldsymbol{k}|,\boldsymbol{k})\overset{\mathscr{L}^{-1}(k)}{\longrightarrow }p^\mu = (p,0,0,p)$$
\end{itemize}
那么 
\[\begin{aligned}
    &\Lambda k = p\\
    &(\Lambda k)^0\to |\boldsymbol{p}|\\
    &W(\Lambda,p)=1\to\sum_{\bar{\lambda}}D_{\lambda \bar{\lambda}}^{j_n}\!\left\{W^{-1}(\Lambda,p)\right\} =1
\end{aligned}\]
所以系数函数选择变为：
\begin{subequations}\label{massless_boost}
    \begin{align}
        &\sum_{\bar\lambda} u_{ab}(\boldsymbol{k},\bar\lambda)\, \sqrt{\frac{k^0}{|\boldsymbol{p}|}}\, \delta_{\lambda\bar\lambda} = \sum_{\bar{a}\bar{b}} \mathscr{D}_{ab;\bar{a}\bar{b}}(\mathscr{L}(k))\, u_{\bar{a}\bar{b}}(\boldsymbol{p},\lambda)\\
        &\sum_{\bar\lambda} v_{ab}(\boldsymbol{k},\bar\lambda)\, \sqrt{\frac{k^0}{|\boldsymbol{p}|}}\, \delta_{\lambda\bar\lambda} = \sum_{\bar{a}\bar{b}} \mathscr{D}_{ab;\bar{a}\bar{b}}(\mathscr{L}(k))\, v_{\bar{a}\bar{b}}(\boldsymbol{p},\lambda)
    \end{align}
\end{subequations}
因此，任意动量处的系数完全由标准动量处的系数 $u(\boldsymbol{p},\lambda)$ 和 $v(\boldsymbol{p},\lambda)$ 决定。
\vspace{2ex}
\begin{center}
    \large\textbf{rotation}  
\end{center}
\vspace{2ex}
前文中，已经解决了$ISO(2)$小群带来的约束问题，现在开始讨论旋转条件。
\begin{itemize}
    \item 令$k^\mu = (k,0,0,k)$，也就是标准系；
    \item 对$ISO(2) $的讨论中；Winger的小群的群元只有$J_3$是有效的；所以选择$\Lambda=J_3$也就是在标准动量下保持$k^\mu=(k,0,0,k)$不变的旋转；
\end{itemize}
那么
\[\begin{aligned}
    &\Lambda k=k\\
    &(\Lambda k)^0\to |\boldsymbol{k}|\\
    &W(\theta,\alpha,\beta)=S(\alpha,\beta)R_3(\theta)=J_3
\end{aligned}\]
由于$\boldmath{J}=\pmb{A}+\pmb{B}$，$Winger$的转动表征为总的自旋$\boldsymbol{J}$,以及(\refeq{massless_lorentz_transf})和上述已经消除了非物理的情况，所以总自旋的表示矩阵为：
\[
D_{\lambda\bar\lambda}^{(j)}(J_3^{-1})=\left[\exp(+iJ_3)\right]_{\lambda\bar\lambda}
\]
场的自旋拆为$\pmb{A}+\pmb{B}$,表示矩阵 $\mathscr{D}(J_3^{-1})$ 也可以分解为 $A$ 和 $B$ 两部分的张量积：
\[
\mathscr{D}(J_3^{-1}) = \exp\left( +i \theta J_3 \right) = \exp\left( +i \theta (A_3 + B_3) \right) = \exp\left( +i \theta A_3 \right) \otimes \exp\left( +i \theta B_3 \right)
\]
在这种分解下交织方程变为：  
\begin{subequations}\label{massless_rotation}
    \begin{align}
        &\sum_{\bar\lambda} u_{ab}(\boldsymbol{k},\bar\lambda)\, e^{-i \theta J_3} \delta_{\lambda\bar\lambda} = \sum_{\bar{a}\bar{b}} \left[ e^{+i \theta A_3} \right]_{a\bar{a}} \left[ e^{+i \theta B_3} \right]_{b\bar{b}} u_{\bar{a}\bar{b}}(\boldsymbol{k},\lambda)\\
        &\sum_{\bar\lambda} v_{ab}(\boldsymbol{k},\bar\lambda)\,e^{+i \theta J_3} \delta_{\lambda\bar\lambda} = \sum_{\bar{a}\bar{b}} \left[ e^{+i \theta A_3} \right]_{a\bar{a}} \left[ e^{+i \theta B_3} \right]_{b\bar{b}} v_{\bar{a}\bar{b}}(\boldsymbol{k},\lambda)
    \end{align}
\end{subequations}
按照5.3的逻辑，以及(\refeq{helcity_o})就是选择$a = -A$ 和 $b = +B$,那么系数函数就是
\begin{equation}
    u_{ab}(\boldsymbol{p},\lambda) =\frac{1}{\sqrt{2|\boldsymbol{p}|}} \langle A,a;B,b \,|\, j, \lambda \rangle=\frac{1}{\sqrt{2|\boldsymbol{p}|}} \langle -A; +B \,|\, j, \lambda \rangle  
\end{equation}
对其做一标准boost $\mathscr{L}(k)$，得到任意动量处的系数函数：
\begin{equation}
    \begin{aligned}
        u_{ab}(\boldsymbol{k},\lambda) = &\sqrt{\frac{|\boldsymbol{p}|}{k^0}} \sum_{\bar{a}\bar{b}} \left[ R(\hat{\boldsymbol{k}}) \right]_{a\bar{a}} \left[ R(\hat{\boldsymbol{k}}) \right]_{b\bar{b}} u_{\bar{a}\bar{b}}(\boldsymbol{k},\lambda)\\
        &= \sqrt{\frac{1}{2k^0}} \sum_{\bar{a}\bar{b}} \left[ R(\hat{\boldsymbol{k}}) \right]_{a\bar{a}} \left[ R(\hat{\boldsymbol{k}}) \right]_{b\bar{b}}  \langle -A; +B \,|\, j, \lambda \rangle\\
    \end{aligned}
\end{equation}
通过(\refeq{massless_rotation})以及(\refeq{massless_boost})，
可以验证这个系数函数满足旋转条件\footnote{准确的说和我们熟悉的对偶场是一个说法}。
\begin{equation}
    v_{ab}(\boldsymbol{k},\lambda) =u_{ab}(\boldsymbol{k},\lambda)^*
\end{equation}
