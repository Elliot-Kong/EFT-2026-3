\section{基本对易关系}
    平移算符\footnote{这里借用后续章节的知识,Poincar\'e代数中的纯平移部分}三维空间形式是：
\begin{equation}
    U(\boldsymbol{a})=e^{-i\boldsymbol{p}\cdot\boldsymbol{a}}
\end{equation}
    作用在位置算符上是:
\begin{equation}
    \begin{aligned}
        &U(\boldsymbol{a})\boldsymbol{x}U^\dagger(\boldsymbol{a})=\boldsymbol{x}-\boldsymbol{a}\\
        \to\quad& x_i + x_i\left(ip_j a^j\right) - \left(ip_j a^j\right)x_i = x_i - a^i \\
        \to\quad& x_i - i a^j (p_j x_i - x_i p_j) = x_i - a^i \\
        \to\quad& x_i - i a^j [p_j, x_i] = x_i - a^i \\
        \to\quad& -i a^j [p_j, x_i] = -a^j\delta_{ij} \\
        \to\quad& -i [p_j, x_i] = -\delta_{ij} \\
        \to\quad& [ x_i,p_j] = i \delta_{ij}
    \end{aligned}
\end{equation}
过渡到经典情况，令$\hbar\to0$\footnote{回到国际单位即可}:\begin{equation}
    [ x_i,p_j] = i \delta_{ij}\overset{\hbar\to0}{\longrightarrow}\{ x_i,p_j\}_{CM}=0
\end{equation}
