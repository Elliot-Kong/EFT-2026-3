% --- 加载 TikZ 库（报错通常是因为缺这个）---
\usetikzlibrary{arrows.meta, decorations.markings, calc, shapes.geometric}
\usetikzlibrary{decorations.markings}
% --- 定义绘图宏 (TikZ Styles) ---
\tikzset{
    % Style 1: 中间带箭头的线 (Mid Arrow)
    % 原理：使用装饰器在路径 55% 处画一个箭头
    mid arrow/.style={
        postaction={decorate},
        decoration={
            markings,
            mark=at position 0.55 with {\arrow{Stealth[length=2mm]}}
        }
    },
    % Style 2: S矩阵/连通部分的圆圈 (Blob)
    % 设置为圆形，白色填充（遮挡背后的线），加粗边缘
    blob/.style={
        circle,
        draw=black,
        thick,
        fill=white,
        minimum size=1.2cm, % 圆圈大小
        inner sep=1pt,
        font=\Large\bfseries % 字体加粗变大
    },
    % Style 3: 外部腿的通用线条样式
    leg/.style={
        thick,
        line cap=round
    }
}
