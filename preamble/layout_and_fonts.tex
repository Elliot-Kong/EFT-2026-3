% ============================================================
% Layout and fonts
% - Page margins (geometry)
% - Header / footer style (fancyhdr)
% - Fonts (fontspec + xeCJK)
%
% Note: Compile with XeLaTeX.
% ============================================================
\usepackage[
  a4paper,
  top=1.5cm,
  bottom=1.5cm,
  left=1.6cm,
  right=1.6cm,
]{geometry}

% Header / footer
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[C]{\thepage}

\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% Keep chapter-opening pages consistent: no header, centered page number in footer.
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% Fonts

\usepackage{fontspec}
\usepackage{xeCJK}
\IfFontExistsTF{Times New Roman}
  {\setmainfont{Times New Roman}}
  {\setmainfont{TeX Gyre Termes}} % TeX Live fallback

% CJK fonts (macOS)
% If you want SimSun, replace `Songti SC` with `SimSun` (and make sure it is installed).
\setCJKmainfont{Songti SC}[
  AutoFakeBold = 5.0,
  AutoFakeSlant = 0.25
]
\usepackage{lmodern}

% A simple sans-serif CJK font
\setCJKsansfont{Heiti SC}
\CJKsetecglue{\hspace{0.3em}} % 增加中西文之间的间距
\usepackage{setspace}
\setstretch{1.15} % 推荐 1.25 到 1.5 之间，物理笔记这类公式多的文档尤其建议加大
\setlength{\parskip}{0.05em} % 设置段落之间的垂直间距
